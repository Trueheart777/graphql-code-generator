var handleNameDuplications = function (name, existing) {
    if (existing.find(function (model) { return model.name === name; })) {
        return '_' + name;
    }
    return name;

exports.buildInnerModelsArray = function (schema, rootObject, selections, primitivesMap, appendTo, result) {
    if (result === void 0) { result = []; }
    (selections ? selections.selections : []).forEach(function (selectionNode) {
        switch (selectionNode.kind) {
            case kinds_1.FIELD:
                var fieldName = selectionNode.name.value;
                var propertyName = selectionNode.alias ? selectionNode.alias.value : fieldName;
                var field = utils_1.getFieldDef(schema, rootObject, selectionNode);
                var rawType = field.type;
                var actualType = definition_1.getNamedType(rawType);
                if (actualType instanceof definition_1.GraphQLObjectType) {
                    var modelName = handleNameDuplications(pascalCase(fieldName), result);
                    var model = {
                        name: modelName,
                        fields: [],
                        fragmentsUsed: [],
                        inlineFragments: []
                    };
                    result.push(model);
                    exports.buildInnerModelsArray(schema, actualType, selectionNode.selectionSet, primitivesMap, model, result);
                    if (!appendTo) {
                        // Means we are on the root object, and we need to create the Result interface
                        appendTo = {
                            name: 'Result',
                            fields: [],
                            fragmentsUsed: [],
                            inlineFragments: []
                        };
                        result.push(appendTo);
                    }
                    appendTo.fields.push({
                        name: propertyName,
                        type: modelName,
                        isArray: model_handler_1.isArray(rawType),
                        isRequired: model_handler_1.isRequired(rawType)
                    });
                }
                else {
                    appendTo.fields.push({
                        name: propertyName,
                        type: model_handler_1.getTypeName(primitivesMap, actualType),
                        isArray: model_handler_1.isArray(rawType),
                        isRequired: model_handler_1.isRequired(rawType)
                    });
                }
                break;
            case kinds_1.FRAGMENT_SPREAD:
                var fragmentName = selectionNode.name.value;
                appendTo.fragmentsUsed.push(pascalCase(fragmentName));
                appendTo.usingFragments = appendTo.fragmentsUsed.length > 0;
                break;
            case kinds_1.INLINE_FRAGMENT:

                var name_1 = selectionNode.typeCondition.name.value + 'InlineFragment';

                break;
        }
    });

};
var getRoot = function (schema, operation) {
    switch (operation.operation) {
        case 'query':
            return schema.getQueryType();
        case 'mutation':
            return schema.getMutationType();
        case 'subscription':
            return schema.getSubscriptionType();
        default:
            return;
    }
};
exports.handleOperation = function (schema, definitionNode, primitivesMap) {
    var name = definitionNode.name.value;
    var type = definitionNode.operation;
    var root = getRoot(schema, definitionNode);
    var builtName = buildName(name, type);
    var document = {
        name: builtName,
        rawName: name,
        isQuery: type === 'query',
        isSubscription: type === 'subscription',
        isMutation: type === 'mutation',
        isFragment: false,
        variables: [],
        innerTypes: [],
        hasVariables: false,
        hasInnerTypes: false,
        imports: [],
        document: printer_1.print(definitionNode)
    };
    document.variables = buildVariables(schema, definitionNode, primitivesMap);
    document.innerTypes = exports.buildInnerModelsArray(schema, root, definitionNode.selectionSet, primitivesMap);
    document.hasVariables = document.variables.length > 0;
    document.hasInnerTypes = document.innerTypes.length > 0;
    document.variables.forEach(function (field) {
        if (field.type && !model_handler_1.isPrimitive(primitivesMap, field.type)) {
            document.imports.push(field.type);
        }
    });
    return document;
};
