"use strict";
var definition_1 = require("graphql/type/definition");
var definition_2 = require("graphql/type/definition");
var introspection_1 = require("graphql/type/introspection");
var definition_3 = require("graphql/type/definition");
var introspection_2 = require("graphql/type/introspection");
var introspection_3 = require("graphql/type/introspection");
function getFieldDef(schema, parentType, fieldAST) {
    var name = fieldAST.name.value;
    if (name === introspection_3.SchemaMetaFieldDef.name &&
        schema.getQueryType() === parentType) {
        return introspection_3.SchemaMetaFieldDef;
    }
    if (name === introspection_2.TypeMetaFieldDef.name &&
        schema.getQueryType() === parentType) {
        return introspection_2.TypeMetaFieldDef;
    }
    if (name === introspection_1.TypeNameMetaFieldDef.name &&
        (parentType instanceof definition_2.GraphQLObjectType ||
            parentType instanceof definition_1.GraphQLInterfaceType ||
            parentType instanceof definition_3.GraphQLUnionType)) {
        return introspection_1.TypeNameMetaFieldDef;
    }
    if (parentType instanceof definition_2.GraphQLObjectType ||
        parentType instanceof definition_1.GraphQLInterfaceType) {
        return parentType.getFields()[name];
    }
}
exports.getFieldDef = getFieldDef;
