exports.isPrimitive = function (primitivesMap, type) {
    return Object.keys(primitivesMap).map(function (key) { return primitivesMap[key]; }).find(function (item) { return item === type; });
};
var shouldSkip = function (typeName) {
    return !typeName ||
        typeName.indexOf('__') > -1 ||
        typeName === 'Query' ||
        typeName === 'Mutation' ||
        typeName === 'Subscription';
};
exports.isRequired = function (type) {
    return (type.toString()).indexOf('!') > -1;
};
exports.isArray = function (type) {
    return (type.toString()).indexOf('[') > -1;
};
exports.getTypeName = function (primitivesMap, type) {
    var name = (type.toString()).replace(/[\[\]!]/g, '');
    if (primitivesMap[name]) {
        return primitivesMap[name];
    }
    else {
        return name;
    }
};
exports.handleType = function (primitivesMap, typeName, type) {
    var currentType = {
        imports: [],
        name: typeName,
        fields: [],
        isEnum: false,
        isObject: false
    };
    if (!shouldSkip(typeName)) {
        if (type instanceof definition_1.GraphQLEnumType) {
            currentType.isEnum = true;
            currentType.enumValues = type.getValues().map(function (enumItem) {
                return {
                    name: enumItem.name,
                    description: enumItem.description,
                    value: enumItem.value
                };
            });
        }
        else if (type instanceof definition_2.GraphQLObjectType || type instanceof definition_3.GraphQLInputObjectType || type instanceof definition_4.GraphQLInterfaceType) {
            currentType.isObject = true;
            var fields_1 = type.getFields();
            if (type instanceof definition_2.GraphQLObjectType) {
                currentType.implementedInterfaces = type.getInterfaces().map(function (interf) {
                    return interf.name;
                });
                currentType.hasImplementedInterfaces = currentType.implementedInterfaces.length > 0;
            }
            currentType.fields = Object
                .keys(fields_1)
                .map(function (fieldName) { return fields_1[fieldName]; })
                .map(function (field) {
                var type = exports.getTypeName(primitivesMap, field.type);
                if (!exports.isPrimitive(primitivesMap, type)) {
                    currentType.imports.push(type);
                }
                return {
                    name: field.name,
                    type: type,
                    isArray: exports.isArray(field.type),
                    isRequired: exports.isRequired(field.type)
                };
            });
        }
        else if (type instanceof definition_5.GraphQLUnionType) {
            currentType.name = type.name || typeName;
            currentType.isUnion = true;
            currentType.isObject = false;
            currentType.unionTypes = type.getTypes().map(function (type) { return type.name; });
            currentType.hasUnionTypes = currentType.unionTypes.length > 0;
        }
        else if (type instanceof definition_6.GraphQLList || type instanceof definition_7.GraphQLNonNull) {
            return exports.handleType(primitivesMap, typeName, definition_8.getNamedType(type));
        }
        return currentType;
    }
    else {
        return null;
    }
};
